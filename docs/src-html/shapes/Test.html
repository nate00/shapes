039  public static void blankwardTest(Direction d) {
040    System.out.println(d.toString() + " is rightward? " + d.isRightward());
041    System.out.println(d.toString() + " is upward? " + d.isUpward());
042    System.out.println(d.toString() + " is leftward? " + d.isLeftward());
043    System.out.println(d.toString() + " is downward? " + d.isDownward());
044  }
045  public static void angleBetweenTest(Direction a, Direction b) {
046    System.out.println("Angle between " + pi(a.getRadians()) + " and " + pi(b.getRadians()) + " is " + pi(Geometry.interiorRadians(a, b)));
047  }
048
049  public static String pi(double z) {
050    return z / Math.PI + " pi";
051  }
052
053  public static void intersectionTest(Segment s, Segment t) {
054    System.out.println("FIRST SEGMENT: " + s.toString());
055    System.out.println("SECOND SEGMENT: " + t.toString());
056    System.out.println("INTERSECTION: " + Geometry.lineIntersection(s, t));
057    System.out.println("INTERSECTION: " + Geometry.segmentIntersection(s, t));
058    System.out.println("--------------------------------------");
059  }
