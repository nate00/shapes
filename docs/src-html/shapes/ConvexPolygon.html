001package shapes;
002
003import java.awt.*;
004
005/**
006 * A convex polygon that appears on screen and interacts with other shapes.
007 * You won't use <code>ConvexPolygon</code> directly. Instead, you'll use one
008 * of its subclasses, {@link Rectangle} or {@link Triangle}.
009 */
010public abstract class ConvexPolygon extends Shape {
011
012  protected boolean displaysRotation;
013  abstract Point[] getUnrotatedCorners();
014
015  public ConvexPolygon() {
016    super();
017    setDisplaysRotation(true);
018  }
019
020  public Point[] getCorners() {
021    Point[] corners = getUnrotatedCorners();
022    if (
023      !displaysRotation ||
024      direction == null ||
025      Math.abs(direction.getRadians()) < Geometry.EPSILON
026    ) {
027      return corners;
028    }
029
030    double angle = direction.getRadians();
031    for (int i = 0; i < corners.length; i++) {
032      double xOffset = corners[i].getX() - center.getX();
033      double yOffset = corners[i].getY() - center.getY();
034      corners[i] = new Point(
035        center.getX() + xOffset * Math.cos(angle) - yOffset * Math.sin(angle),
036        center.getY() + xOffset * Math.sin(angle) + yOffset * Math.cos(angle)
037      );
038    }
039
040    return corners;
041  }
042
043  double maxRadius() {
044    double max = 0;
045    for (Point corner : getCorners()) {
046      max = Math.max(max, center.distanceTo(corner));
047    }
048    return max;
049  }
050
051  Segment[] getSides() {
052    Point[] corners = getCorners();
053    Segment[] sides = new Segment[corners.length];
054    for (int i = 0; i < corners.length; i++) {
055      sides[i] = new Segment(corners[i], corners[(i + 1) % corners.length]);
056    }
057    return sides;
058  }
059
060  Point maxMovement(Point target, Segment obstacle) {
061    Segment path = new Segment(getCenter(), target);
062    Point maxMove = Geometry.maxMovement(this, target, obstacle);
063    maxMove = Geometry.insertGap(this, path, maxMove);
064    return maxMove;
065  }
066
067  Point maxMovement(Point target, Shape obstacle) {
068    Segment path = new Segment(getCenter(), target);
069    Point maxMove = Geometry.maxMovement(this, target, obstacle);
070    maxMove = Geometry.insertGap(this, path, maxMove);
071    return maxMove;
072  }
073
074  public boolean isOffscreen() {
075    for (Point corner : getCorners()) {
076      if (!Geometry.offscreen(corner)) {
077        return false;
078      }
079    }
080    return true;
081  }
082
083  public boolean contains(Shape shape) {
084    if (this.isDestroyed() || shape.isDestroyed()) {
085      return false;
086    }
087    if (shape instanceof ConvexPolygon) {
088      Point[] corners = ((ConvexPolygon) shape).getCorners();
089      for (Point corner : corners) {
090        if (!this.contains(corner)) {
091          return false;
092        }
093      }
094      return true;
095    } else if (shape instanceof Circle) {
096      Circle circle = (Circle) shape;
097      if (!this.contains(circle.getCenter())) {
098        return false;
099      }
100      for (Segment side : getSides()) {
101        double distance =
102          Geometry.perpendicularThrough(side, circle.getCenter()).length();
103        if (distance < circle.getRadius()) {
104          return false;
105        }
106      }
107      return true;
108    }
109    return false;
110  }
111
112  public boolean contains(Point p) {
113    // the point is inside if all cross products have the same sign
114    boolean positive = false;
115    boolean first = true;
116    for (Segment side : getSides()) {
117      boolean pos =
118        Geometry.cross(side.vector(), new Vector(side.getStart(), p)) > 0;
119      if (first) {
120        positive = pos;
121        first = false;
122        continue;
123      }
124      if (pos != positive) {
125        return false;
126      }
127    }
128    return true;
129  }
130
131  void renderSpeech(Graphics2D g) {
132    if (!isSpeaking()) {
133      return;
134    }
135    g.setColor(getSpeechColor());
136    // upper left of the talk bubble - upper right of the polygon
137    Point upperLeft = null;
138    for (Point corner : getCorners()) {
139      if (upperLeft == null) {
140        upperLeft = corner;
141        continue;
142      }
143      upperLeft.setX(Math.max(upperLeft.getX(), corner.getX()));
144      upperLeft.setY(Math.max(upperLeft.getY(), corner.getY()));
145    }
146    Vector heightOffset =
147      new Vector(0.0, -1.0 * g.getFontMetrics(g.getFont()).getDescent());
148
149    Point lowerLeft = upperLeft.translation(heightOffset);
150    g.drawString(
151      getSpeech(),
152      (int)lowerLeft.getCanvasX(),
153      (int)lowerLeft.getCanvasY()
154    );
155  }
156
157  void render(Graphics2D g) {
158    renderSpeech(g);
159    if (this.isInvisible()) {
160      return;
161    }
162
163    g.setColor(getColor());
164    Point[] corners = getCorners();
165    int[] x = new int[corners.length];
166    int[] y = new int[corners.length];
167    for (int i = 0; i < corners.length; i++) {
168      x[i] = (int)corners[i].getCanvasX();
169      y[i] = (int)corners[i].getCanvasY();
170    }
171    if (this.isFilled()) {
172      g.fillPolygon(x, y, corners.length);
173    } else {
174      g.drawPolygon(x, y, corners.length);
175    }
176  }
177
178  public double getRight() {
179    double right = Double.NEGATIVE_INFINITY;
180    for (Point corner : getCorners()) {
181      right = Math.max(right, corner.getX());
182    }
183    return right;
184  }
185  public double getTop() {
186    double top = Double.NEGATIVE_INFINITY;
187    for (Point corner : getCorners()) {
188      top = Math.max(top, corner.getY());
189    }
190    return top;
191  }
192  public double getLeft() {
193    double left = Double.POSITIVE_INFINITY;
194    for (Point corner : getCorners()) {
195      left = Math.min(left, corner.getX());
196    }
197    return left;
198  }
199  public double getBottom() {
200    double bottom = Double.POSITIVE_INFINITY;
201    for (Point corner : getCorners()) {
202      bottom = Math.min(bottom, corner.getY());
203    }
204    return bottom;
205  }
206
207  /**
208   * Set whether this shape displays the direction it is facing. When set to
209   * false, calls to {@link #setDirection} will not change the appearance of
210   * this shape, but they will still affect which direction this shape moves
211   * when {@link #move(double)} is called.
212   *
213   * @param displaysRotation  true to display this shape's rotation, false to
214   *                          hide it.
215   */
216  public void setDisplaysRotation(boolean displaysRotation) {
217    this.displaysRotation = displaysRotation;
218  }
219
220  public boolean displaysRotation() {
221    return displaysRotation;
222  }
223}
