001package shapes;
002
003import java.awt.*;
004import java.awt.geom.*;
005
006/**
007 * Used to change the visual style of text displayed in the game. Can be used
008 * to customize text's font, size, color, etc.
009 * <p>
010 * There are a few places you can display text, including {@link Shape#say},
011 * {@link Game#setTitle}, {@link Game#setSubtitle}, and {@link Counter}.
012 * <p>
013 * <strong>Example usage:</strong>
014 * <p>
015 * <code>
016 *  TextStyle speechStyle = new TextStyle("Helvetica", 12, Color.BLACK);<br />
017 *  circle.setSpeechStyle(speechStyle);<br />
018 *  circle.say("Hello, world!");
019 * </code>
020 */
021public class TextStyle {
022  private String fontName;
023  private int fontSize;
024  private Color color;
025  private Color backgroundColor;
026  private boolean bold;
027  private boolean italic;
028
029  enum ReferencePointLocation { CENTER, BOTTOM_LEFT, TOP_LEFT, BOTTOM_CENTER };
030
031  public TextStyle(String fontName, int fontSize, Color color) {
032    setFontName(fontName);
033    setFontSize(fontSize);
034    setColor(color);
035    setBold(bold);
036    setItalic(italic);
037  }
038
039  void applyTo(Graphics2D g) {
040    g.setColor(color);
041    g.setFont(getFont());
042  }
043
044  Font getFont() {
045    int style;
046    if (!bold && !italic) {
047      style = Font.PLAIN;
048    } else {
049      style = (bold ? Font.BOLD : 0) | (italic ? Font.ITALIC : 0);
050    }
051    
052    return new Font(fontName, style, fontSize);
053  }
054
055  void renderString(
056    String string,
057    Point referencePoint,
058    ReferencePointLocation referenceLocation,
059    Graphics2D g,
060    Point speechOrigin
061  ) {
062    int boxMargin = 3;
063
064    // find height and width of rendered speech
065    FontMetrics metrics = g.getFontMetrics(getFont());
066    String[] lines = string.split("\n");
067    double width = 0.0;
068    for (String line : lines) {
069      width = Math.max(width, metrics.stringWidth(string));
070    }
071    // getAscent() includes room for accents, etc., so we shrink it (by
072    // an arbitrary amount)
073    double wordHeight = metrics.getAscent() * 0.8;
074    double spaceHeight = metrics.getHeight() - wordHeight;
075    double height =
076      (lines.length - 1) * spaceHeight + lines.length * wordHeight;
077    Vector offset = null; // from reference point to bottom left
078    if (backgroundColor != null) {
079      width += 2 * boxMargin;
080      height += 2 * boxMargin;
081    }
082    switch (referenceLocation) {
083      case CENTER:
084        offset = new Vector(width / -2.0, height / -2.0);
085        break;
086      case BOTTOM_LEFT:
087        offset = new Vector(0, 0);
088        break;
089      case TOP_LEFT:
090        offset = new Vector(0, -1.0 * height);
091        break;
092      case BOTTOM_CENTER:
093        offset = new Vector(width / -2.0, 0);
094        break;
095    }
096    Point bottomLeft = referencePoint.translation(offset);
097    Point textBottomLeft = bottomLeft;
098    Vector lineOffset = new Vector(0, wordHeight + spaceHeight);
099
100    // render background
101    if (backgroundColor != null) {
102      height += metrics.getDescent();
103      Point topLeft = bottomLeft.translation(new Vector(0, height));
104
105      g.setColor(backgroundColor);
106      g.fillRect(
107        topLeft.getCanvasX(),
108        topLeft.getCanvasY(),
109        (int) width,
110        (int) height
111      );
112
113      textBottomLeft = bottomLeft.translation(
114        new Vector(boxMargin, boxMargin + metrics.getDescent())
115      );
116
117      // render speech bubble "foot"
118      if (speechOrigin != null) {
119        Point third = bottomLeft.translation(new Vector(7, 0));  // TODO: height?
120        int[] x = new int[] {
121          bottomLeft.getCanvasX(),
122          speechOrigin.getCanvasX(),
123          third.getCanvasX()
124        };
125        int[] y = new int[] {
126          bottomLeft.translation(new Vector(0, 1)).getCanvasY(),
127          speechOrigin.getCanvasY(),
128          // translate upward to eliminate gap between foot and main background
129          // caused by rounding
130          third.translation(new Vector(0, 1)).getCanvasY()
131        };
132
133        g.fillPolygon(x, y, 3);
134      }
135    }
136
137    // render text
138    applyTo(g);
139    for (int i = lines.length - 1; i >= 0; i--) {
140      g.drawString(
141        lines[i],
142        textBottomLeft.getCanvasX(),
143        textBottomLeft.getCanvasY()
144      );
145      bottomLeft = bottomLeft.translation(lineOffset);
146    }
147  }
148
149  public static TextStyle monospaced() {
150    Font font = new Font(Font.MONOSPACED, Font.PLAIN, 16);
151    return new TextStyle(font.getFontName(), 16, Color.BLACK);
152  }
153
154  public static TextStyle serif() {
155    Font font = new Font(Font.SERIF, Font.PLAIN, 16);
156    return new TextStyle(font.getFontName(), 16, Color.BLACK);
157  }
158  
159  public static TextStyle sansSerif() {
160    Font font = new Font(Font.SANS_SERIF, Font.PLAIN, 16);
161    return new TextStyle(font.getFontName(), 16, Color.BLACK);
162  }
163  
164  /* Getters and setters */
165
166  public void setFontName(String fontName) {
167    if (fontName == null) {
168      throw new IllegalArgumentException("fontName must not be null.");
169    }
170    this.fontName = fontName;
171  }
172
173  public String getFontName() {
174    return fontName;
175  }
176
177  public void setFontSize(int fontSize) {
178    if (fontSize <= 0) {
179      throw new IllegalArgumentException("fontSize must be positive");
180    }
181    this.fontSize = fontSize;
182  }
183
184  public int getFontSize() {
185    return fontSize;
186  }
187
188  public void setColor(Color color) {
189    if (color == null) {
190      throw new IllegalArgumentException("color must not be null.");
191    }
192    this.color = color;
193  }
194
195  public Color getColor() {
196    return color;
197  }
198
199  public void setBold(boolean bold) {
200    this.bold = bold;
201  }
202
203  public boolean isBold() {
204    return bold;
205  }
206
207  public void setItalic(boolean italic) {
208    this.italic = italic;
209  }
210
211  public boolean isItalic() {
212    return italic;
213  }
214
215  // set null for no background
216  public void setBackgroundColor(Color background) {
217    this.backgroundColor = background;
218  }
219
220  public Color getBackgroundColor() {
221    return backgroundColor;
222  }
223}
