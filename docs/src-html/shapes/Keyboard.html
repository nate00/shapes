007/**
008 * Captures input from the keyboard. <code>Keyboard</code>, along with
009 * {@link Mouse}, is how you make your game interactive.
010 * <p>
011 * For example, to make a shape move according to the arrow keys, you would
012 * write:
013 * <p>
014 * <code>shape.move(Keyboard.direction(), 10);</code>
015 */
016public class Keyboard extends KeyAdapter {
017  private static Set<Integer> keysPressed;
018  private static Integer mostRecentKeyPressed;
019  private static boolean printKeyPresses = false;
020
021  Keyboard() {
022    if (keysPressed == null) {
023      keysPressed =
024        Collections.newSetFromMap(new ConcurrentHashMap<Integer, Boolean>());
025    }
026  }
027
028  /**
029   * Print keypresses to standard output. Most of the methods in this class use
030   * codes to represent keys, so this method can be used to determine which
031   * codes represent which keys.
032   * <p>
033   * This class has constants equal to the codes for most common keys. For
034   * example,
035   * <code>boolean upIsPressed = Keyboard.keyIsPressed(Keyboard.UP);</code> can
036   * be used to determine whether the up arrow is pressed.
037   *
038   * @param printKeyPresses passing <code>true</code> prints all key presses.
039   */
040  public static void printKeyPresses(boolean printKeyPresses) {
041    Keyboard.printKeyPresses = printKeyPresses;
042  }
043
044  /**
045   * Returns true if the given key is being pressed.
046   *
047   * @param keyCode integer representing the key whose presses to detect. (See
048   *                {@link #printKeyPresses} to find which strings represent
049   *                which keys.)
050   * @return        true if the key is being pressed, false otherwise.
051   */
052  public static boolean keyIsPressed(int keyCode) {
053    for (int pressedCode : keysPressed) {
054      if (keyCode == pressedCode) {
055        return true;
056      }
057    }
058    return false;
059  }
060
061  /**
062   * Returns an array of integers representing all keys currently being pressed.
063   * Returns an empty array if no keys are being pressed. See
064   * {@link #printKeyPresses} to find which codes represent which keys.
065   *
066   * @return  an array of integers representing all keys currently being pressed.
067   */
068  public static int[] keys() {
069    Iterator<Integer> keyPressedIterator = keysPressed.iterator();
070    int[] keys = new int[keysPressed.size()];
071    for (int i = 0; i < keys.length && keyPressedIterator.hasNext(); i++) {
072      keys[i] = keyPressedIterator.next();
073    }
074
075    return keys;
076  }
077
078  /**
079   * Returns an integer code representing the most recently pressed key.
080   * Returns <code>Keyboard.NO_KEY</code>
081   * if no key is currently being pressed. See {@link #printKeyPresses} to find
082   * which integer codes represent which keys.
083   *
084   * @return  an integer code representing the most recently pressed key.
085   */
086  public static int key() {
087    return mostRecentKeyPressed;
088  }
089
090  /**
091   * Returns the direction currently being pressed on the arrow keys. Returns
092   * null if no direction is being pressed (or if the directions being pressed
093   * cancel each other out, like left and right).
094   *
095   * @return  a <code>Direction</code> object representing the direction being
096   *          pressed on the arrow keys.
097   */
098  public static Direction direction() {
099    return direction(KeySet.ARROWS);
100  }
101
102  /**
103   * Returns the direction currently being pressed on the given key set.
104   * Returns null if no direction is being pressed (or if the directions being
105   * pressed cancel each other out, like left and right).
106   *
107   * @param set the set of keys to get directions from, either
108   *            {@link KeySet#ARROWS} or {@link KeySet#WASD}.
109   * @return    a <code>Direction</code> object representing the direction
110   *            being pressed on the given key set.
111   */
112  public static Direction direction(KeySet set) {
113    if (set == null) {
114      throw new IllegalArgumentException("set must not be null.");
115    }
116    ArrayList<Vector> vectorsPressed = new ArrayList<Vector>();
117    for (int keyPressed : keysPressed) {
118      Vector vectorPressed = set.getVector(keyPressed);
119      if (vectorPressed != null) {
120        vectorsPressed.add(vectorPressed);
121      }
122    }
123
124    if (vectorsPressed.size() == 0) {
125      return null;
126    }
127    Vector ret = vectorsPressed.get(0);
128    for (int i = 1; i < vectorsPressed.size(); i++) {
129      ret = ret.add(vectorsPressed.get(i));
130    }
131    if (Math.abs(ret.getMagnitude()) < Geometry.EPSILON) {
132      return null;
133    }
134    return ret.getDirection();
135  }
136
137  /**
138   * You can ignore this method. This method gets called whenever a key is
139   * pressed.
140   */
141  @Override
142  public void keyPressed(KeyEvent keyPressedEvent) {
143    keysPressed.add(keyPressedEvent.getKeyCode());
144    mostRecentKeyPressed = keyPressedEvent.getKeyCode();
145    if (printKeyPresses) {
146      System.out.println(keyPressedEvent.getKeyCode());
147    }
148  }
149
150  /**
151   * You can ignore this method. This method gets called whenever a key is
152   * released.
153   */
154  @Override
155  public void keyReleased(KeyEvent keyReleasedEvent) {
156    // remove from keysPressed
157    Iterator<Integer> iter = keysPressed.iterator();
158    while (iter.hasNext()) {
159      if (iter.next() == keyReleasedEvent.getKeyCode()) {
160        iter.remove();
161      }
162    }
163
164    if (mostRecentKeyPressed == keyReleasedEvent.getKeyCode()) {
165      mostRecentKeyPressed = NO_KEY;
166    }
167  }
