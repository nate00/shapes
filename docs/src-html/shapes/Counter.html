001package shapes;
002
003import java.awt.*;
004import java.util.*;
005
006/**
007 * Displays numerical values, such as score or lives remaining. Counters are
008 * displayed in the upper-left corner of the game window. They have a name and
009 * value, both of which are displayed.
010 */
011public class Counter {
012  private String name;
013  private int value;
014  private boolean destroyed = false;
015
016  /**
017   * Construct a new counter with a given name and value.
018   *
019   * @param name          the name of this counter.
020   * @param initialValue  this counter's starting numerical value.
021   */
022  public Counter(String name, int initialValue) {
023    if (name == null) {
024      throw new IllegalArgumentException("name must not be null.");
025    }
026    this.name = name;
027    this.value = initialValue;
028    Game.addCounter(this);
029  }
030
031  /**
032   * Increase this counter's value by a given amount.
033   *
034   * @param amount amount by which to increase this counter's value.
035   */
036  public void increaseBy(int amount) {
037    value += amount;
038  }
039
040  /**
041   * Decrease this counter's value by a given amount.
042   *
043   * @param amount  amount by which to decrease this counter's value.
044   */
045  public void decreaseBy(int amount) {
046    value -= amount;
047  }
048
049  /**
050   * Destroys this counter so it will be removed from the game. Call this method
051   * only when you have finished using the counter.
052   * <p>
053   * A destroyed counter will not appear on the screen. A destroyed counter
054   * cannot be undestroyed.
055   */
056  // TODO: test whether this works
057  public void destroy() {
058    destroyed = true;
059    Game.removeCounter(this);
060  }
061
062  static void renderCounters(java.util.List<Counter> counters, TextStyle style, Graphics2D g) {
063    Point windowTopLeft = new Point(0, Game.HEIGHT);
064    Point countersTopLeft = windowTopLeft.translation(new Vector(10, -10));
065    StringBuffer buf = new StringBuffer();
066    Iterator<Counter> iter = counters.iterator();
067    while (iter.hasNext()) {
068      buf.append(iter.next().toString());
069      if (iter.hasNext()) {
070        buf.append('\n');
071      }
072    }
073    style.renderString(
074      buf.toString(),
075      countersTopLeft,
076      TextStyle.ReferencePointLocation.TOP_LEFT,
077      g,
078      null
079    );
080  }
081
082  /* Getters and settings*/
083
084  /**
085   * Sets this counter's name.
086   *
087   * @param name  this counter's new name.
088   */
089  public void setName(String name) {
090    if (name == null) {
091      throw new IllegalArgumentException("name must not be null.");
092    }
093    this.name = name;
094  }
095
096  /**
097   * Returns this counter's name.
098   *
099   * @return this counter's name.
100   */
101  public String getName() {
102    return name;
103  }
104
105  /**
106   * Set this counter's numerical value.
107   *
108   * @param value this counter's new numerical value.
109   */
110  public void setValue(int value) {
111    this.value = value;
112  }
113
114  /**
115   * Returns this counter's numerical value.
116   *
117   * @return this counter's numerical value.
118   */
119  public int getValue() {
120    return value;
121  }
122
123  /**
124   * Return whether this counter has been destroyed.
125   *
126   * @return  <code>true</code> if this counter has been destroyed,
127   *          <code>false</code> if not.
128   */
129  public boolean isDestroyed() {
130    return destroyed;
131  }
132
133  /**
134   * Returns a string representing this counter.
135   *
136   * @return  a string representing this counter.
137   */
138  @Override
139  public String toString() {
140    return name + "  " + value;
141  }
142}
