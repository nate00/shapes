003// TODO: debugging
004import java.awt.*;
005
006/**
007 * Represents a location in the 2D plane. Points are not displayed in the game.
008 */
009// TODO: debugging
010public class Point //extends Shape {
011{
012  private double x, y;
013
014  public Point(double x, double y) {
015    this.x = x;
016    this.y = y;
017  }
018
019  public Point translation(Vector v) {
020    if (v == null) {
021      throw new IllegalArgumentException("direction must not be null.");
022    }
023    Point p = new Point(x, y);
024    p.setX(p.getX() + v.getXComponent());
025    p.setY(p.getY() + v.getYComponent());
026    return p;
027  }
028
029  public Point translation(Direction direction, double pixels) {
030    return translation(new Vector(direction, pixels));
031  }
032
033  public double getX() {
034    return x;
035  }
036
037  public double getY() {
038    return y;
039  }
040  
041  int getCanvasX() {
042    return (int)x;
043  }
044
045  int getCanvasY() {
046    return (int)(Game.HEIGHT - y);
047  }
048
049  public void setX(double x) {
050    this.x = x;
051  }
052
053  public void setY(double y) {
054    this.y = y;
055  }
056
057  public static Point random() {
058    double x = Math.random() * Game.WIDTH;
059    double y = Math.random() * Game.HEIGHT;
060    return new Point(x, y);
061  }
062
063  public double distanceTo(Point point) {
064    return Geometry.distance(this, point);
065  }
066
067  @Override
068  public boolean equals(Object obj) {
069    if (obj instanceof Point) {
070      return this.equals((Point) obj);
071    }
072
073    return false;
074  }
075
076  private boolean equals(Point that) {
077    return 
078      Math.abs(this.getX() - that.getX()) < Geometry.TOLERANCE &&
079      Math.abs(this.getY() - that.getY()) < Geometry.TOLERANCE;
080  }
081
082  @Override
083  public String toString() {
084    return String.format("(%f, %f)", x, y);
085  }
086
087  // TODO: for debugging
088  public Point maxMovement(Point p, Shape s) { return Point.random(); }
089  public boolean isOffscreen() { return false; }
090  public boolean contains(Point p) { return false; }
091  public boolean contains(Shape s) { return false; }
092  public void render(Graphics2D g) {
093    if (!visible) return;
094    g.setColor(Color.YELLOW);
095    g.fillOval((int)getCanvasX() - 2, (int)getCanvasY() - 2, 4, 4);
096  }
097  private boolean visible = false;
098  public void update() { }
099  public void setup() { }
100  public void setVisible() {visible = true;}
101  public double getBottom() {return 0;}
102  public double getTop() {return 0;}
103  public double getLeft() {return 0;}
104  public double getRight() {return 0;}
105  public Point maxMovement(Point p, Segment s) {return p;}
106
107}
