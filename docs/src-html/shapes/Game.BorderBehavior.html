015/**
016 * A game built using the Shapes framework.
017 * <p>
018 * <code>Game</code> is one of the two most important classes (along with
019 * {@link Shape}) for building a game using the Shapes framework. To make
020 * a game, you will subclass <code>Game</code> and override the
021 * {@link #setup()} and {@link #update()} functions. In <code>setup()</code>,
022 * you will write code that will execute when your game begins, and in
023 * <code>update()</code> you will write code that executes once per frame.
024 * <p>
025 * <code>Game</code> also has a few useful static methods and constants. For
026 * example, you can call <code>Game.getAllShapes()</code> to get all the shapes
027 * that currently exist in your game.
028 */
029public abstract class Game {
030  private static JFrame frame;
031  private static Canvas canvas;
032  private static Set<Shape> solidShapes;
033  private static Set<Shape> allShapes;
034
035  private static Map<Integer, java.util.List<Shape>> layerContents;
036  private static java.util.List<Integer> layers;
037  private static Map<Shape, Integer> layerOf;
038
039  private static String title;
040  private static int titleDuration;
041
042  public enum BorderBehavior { NONE, SOLID, BOUNCE };
043  private static BorderBehavior borderBehavior;

045  public static final int HEIGHT = 500;
046  public static final int WIDTH = 800;

048  public Game() {
049    canvas = new Canvas(this);
050    solidShapes =
051      Collections.newSetFromMap(new ConcurrentHashMap<Shape, Boolean>());
052    allShapes = 
053      Collections.newSetFromMap(new ConcurrentHashMap<Shape, Boolean>());
054
055    // TODO: sort out which data structures actually have to support concurrency
056    layerContents = new ConcurrentHashMap<Integer, java.util.List<Shape>>();
057    layers = new CopyOnWriteArrayList<Integer>();
058    layerOf = new ConcurrentHashMap<Shape, Integer>();
059
060    frame = new JFrame();
061    Mouse mouse = new Mouse();
062    frame.addMouseMotionListener(mouse);
063    frame.addMouseListener(mouse);
064    frame.addKeyListener(new Keyboard());
065    frame.setDefaultCloseOperation(frame.EXIT_ON_CLOSE);
066
067    setBackgroundColor(Color.BLUE);
068    setBorderBehavior(BorderBehavior.NONE);
069//    setup();
070
071  }

089  void autoUpdate() {
090    for (Shape s : allShapes) {
091      s.autoUpdate();
092    }
093
094    Iterator<Shape> iter = allShapes.iterator();
095    while (iter.hasNext()) {
096      Shape s = iter.next();
097      s.update();
098      if (s.isDestroyed()) {
099        if (s.isSolid()) {
100          removeSolid(s);
101        }
102        removeFromLayers(s);
103        iter.remove();
104      }
105    }
106  }

125    allShapes.remove(shape);

127  }

129  public static Shape[] getAllShapes() {
130    return allShapes.toArray(new Shape[0]);
131  }
132
133  static void removeFromLayers(Shape shape) {
134    if (!layerOf.containsKey(shape)) return;
135
136    int oldLayer = layerOf.get(shape);
137    layerContents.get(oldLayer).remove(shape);
138    if (layerContents.get(oldLayer).isEmpty()) {
139      layerContents.remove(oldLayer);
140      layers.remove((Integer) oldLayer);
141    }
142    layerOf.remove(shape);
143  }
144
145  static void setLayer(Shape shape, int layer) {
146    removeFromLayers(shape);
147
148    // add new stuff
149    if (!layerContents.containsKey(layer)) {
150      layerContents.put(layer, new CopyOnWriteArrayList<Shape>());
151      int insertionPoint = ~Collections.binarySearch(layers, layer);
152      layers.add(insertionPoint, layer);
153    }
154    layerContents.get(layer).add(shape);
155    layerOf.put(shape, layer);
156  }
157
158  static java.util.List<Shape> getLayerContents(int layer) {
159    return layerContents.get(layer);
160  }
161
162  static int getLayerOf(Shape shape) {
163    return layerOf.get(shape);
164  }
165
166  static java.util.List<Integer> getLayers() {
167    return layers;
168  }

195  public static void setTitle(String title, int duration) {
196    if (duration < 0) {
197      throw new IllegalArgumentException("Duration cannot be negative.");
198    }
199    if (title == null) {
200      throw new IllegalArgumentException("Title cannot be null.");
201    }
202    Game.title = title;
203    Game.titleDuration = duration;
204  }

206  void renderTitle(Graphics2D g) {
207    titleDuration--;
208    g.setFont(new Font("Times New Roman", Font.PLAIN, 60));
209    g.setColor(Color.BLACK);
210    Rectangle2D containingBox =
211      g.getFontMetrics(g.getFont()).getStringBounds(title, g);
212    double height = containingBox.getHeight();
213    double width = containingBox.getWidth();
214    Point bottomLeft = new Point(
215      (WIDTH - width) / 2.0,
216      HEIGHT / 2.0
217    );
218    System.out.println(bottomLeft.toString());
219    g.drawString(title, bottomLeft.getCanvasX(), bottomLeft.getCanvasY());
220  }

222  /**
223   * Set the behavior of the window borders. The options are:
224   * <ul>
225   *  <li><code>NONE</code>: allow shapes to leave the window.</li>
226   *  <li><code>SOLID</code>: prevent shapes from leaving the window. (See
227   *  {@link Shape#setSolid}.)</li>
228   *  <li><code>BOUNCE</code>: make shapes bounce when they hit the window
229   *  borders, i.e., reverse direction as if bouncing off a physical
230   *  border.</li>
231   * </ul>
232   * <p>
233   * Syntax example:
234   * <code>
235   *  Game.setBorderBehavior(Game.BorderBehavior.SOLID);
236   * </code>
237   *
238   * @param behavior  specifies how shapes will act when they reach the border
239   *                  of the window.
240   */
241  public static void setBorderBehavior(BorderBehavior behavior) {
242    Game.borderBehavior = behavior;
243  }

245  /**
246   * Returns the behavior of the border windows. The options are:
247   * <ul>
248   *  <li><code>NONE</code>: allow shapes to leave the window.</li>
249   *  <li><code>SOLID</code>: prevent shapes from leaving the window. (See
250   *  {@link Shape#setSolid}.)</li>
251   *  <li><code>BOUNCE</code>: make shapes bounce when they hit the window
252   *  borders, i.e., reverse direction as if bouncing off a physical
253   *  border.</li>
254   * </ul>
255   *
256   * @return  a <code>BorderBehavior</code> specifying the way shapes react
257   *          when they reach the border of the window.
258   */
259  public static BorderBehavior getBorderBehavior() {
260    return borderBehavior;
261  }

263  public enum Border { 
264    TOP(Direction.UP, new Segment(0, HEIGHT, WIDTH, HEIGHT)),
265    RIGHT(Direction.RIGHT, new Segment(WIDTH, HEIGHT, WIDTH, 0)),
266    BOTTOM(Direction.DOWN, new Segment(WIDTH, 0, 0, 0)),
267    LEFT(Direction.LEFT, new Segment(0, 0, 0, HEIGHT)),
268    OFFSCREEN(null, null);
269
270    private Direction direction;
271    private Segment segment;
272  
273    private Border(Direction d, Segment seg) {
274      direction = d;
275      segment = seg;
276    }
277
278    public Direction getDirection() {
279      return direction;
280    }
281
282    Segment getSegment() {
283      return segment;
284    }
285
286    public static Border[] all() {
287      return new Border[] { TOP, RIGHT, BOTTOM, LEFT };
288    }
289  };
