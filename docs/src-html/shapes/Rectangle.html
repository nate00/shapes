001package shapes;
002
003import java.awt.*;
004
005/**
006 * A rectangle that appears on screen and interacts with other shapes.
007 * Many of <code>Rectangles</code>'s useful methods are in its superclasses
008 * {@link Shape} and {@link ConvexPolygon}.
009 */
010public class Rectangle extends ConvexPolygon {
011  private double width, height;
012
013  public Rectangle(Point center, double width, double height) {
014    setCenter(center);
015    setWidth(width);
016    setHeight(height);
017
018    // set defaults
019    setColor(Color.GREEN);
020  }
021
022  public Rectangle() {
023    this(new Point(100, 100), 50, 100);
024  }
025
026  // Override this!
027  public void setup() {}
028
029  // Override this!
030  public void update() {}
031
032  Point[] getUnrotatedCorners() {
033    return new Point[] {
034      getCenter().translation(new Vector(-1 * width / 2, height / 2)),
035      getCenter().translation(new Vector(width / 2,      height / 2)),
036      getCenter().translation(new Vector(width / 2,      -1 * height / 2)),
037      getCenter().translation(new Vector(-1 * width / 2, -1 * height / 2))
038    };
039  }
040
041  @Override
042  double maxRadius() {
043    return Geometry.hypoteneuse(width / 2, height / 2);
044  }
045
046  public void setUpperLeft(Point upperLeft) {
047    if (upperLeft == null) {
048      throw new IllegalArgumentException("upperLeft must not be null.");
049    }
050    setCenter(upperLeft.getX() + width / 2, upperLeft.getY() - height / 2);
051  }
052
053  public double getHeight() {
054    return height;
055  }
056
057  public void setHeight(double height) {
058    this.height = height;
059  }
060
061  public double getWidth() {
062    return width;
063  }
064
065  public void setWidth(double width) {
066    this.width = width;
067  }
068
069  @Override
070  public String toString() {
071    return "Rectangle at " + getCenter().toString() +
072      " with height " + getHeight() + " and width " + getWidth();
073  }
074}
