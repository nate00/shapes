001package shapes;
002
003import java.awt.*;
004import java.awt.event.*;
005
006/**
007 * Captures input from the mouse. <code>Mouse</code>, along with
008 * {@link Keyboard}, is how you make your game interactive.
009 * <p>
010 * For example, to make a shape move towards the point where the mouse is
011 * clicked, you would write:
012 * <p>
013 * <code>shape.move(shape.towards(Mouse.clickLocation(), 10);</code>
014 */
015public class Mouse implements MouseMotionListener, MouseListener {
016  private static Point click;
017  private static Point mouse;
018  private static boolean inWindow = false;
019  
020  /**
021   * Returns the location of a mouse click. If the mouse button is pressed and
022   * then the mouse is dragged (without releasing the button), the new location
023   * of the mouse is returned. Unlike {@link #mouseLocation}, this function
024   * will return <code>null</code> if the mouse button isn't pressed.
025   *
026   * @return  the location of the mouse cursor, if the mouse button is pressed.
027   *          Returns <code>null</code> if the mouse button isn't pressed.
028   */
029  public static Point clickLocation() {
030    return click;
031  }
032
033  /**
034   * Returns the location of the mouse cursor. Unlike {@link #clickLocation},
035   * this function will return the mouse's location even if the mouse button
036   * isn't pressed.
037   *
038   * @return  the location of the mouse cursor. If the mouse cursor hasn't yet
039   *          entered the game window, returns <code>null</code>. If the cursor
040   *          was in the window but has left, returns the location of the mouse
041   *          just before it left the window.
042   */
043  public static Point mouseLocation() {
044    return mouse;
045  }
046
047  /**
048   * Returns whether the mouse cursor is in the game window.
049   *
050   * @return  <code>false</code> if the mouse cursor is outside the game window,
051   *          <code>true</code> if inside.
052   */
053  public static boolean isInWindow() {
054    return inWindow;
055  }
056
057  private void updateClick(MouseEvent event) {
058    click = new Point(
059      Geometry.fromCanvasX(event.getX()),
060      Geometry.fromCanvasY(event.getY())
061    );
062  }
063
064  private void updateMouse(MouseEvent event) {
065    mouse = new Point(
066      Geometry.fromCanvasX(event.getX()),
067      Geometry.fromCanvasY(event.getY())
068    );
069  }
070
071  /**
072   * You can ignore this method. This method gets called when the mouse
073   * button is pressed, and is used internally.
074   */
075  public void mousePressed(MouseEvent event) {
076    updateClick(event);
077    updateMouse(event);
078  }
079
080  /**
081   * You can ignore this method. This method gets called when the mouse
082   * is dragged, and is used internally.
083   */
084  public void mouseDragged(MouseEvent event) {
085    updateClick(event);
086    updateMouse(event);
087  }
088
089  /**
090   * You can ignore this method. This method gets called when the mouse
091   * button is released, and is used internally.
092   */
093  public void mouseReleased(MouseEvent event) {
094    click = null;
095  }
096
097  /**
098   * You can ignore this method. This method gets called when the mouse
099   * is moved, and is used internally.
100   */
101  public void mouseMoved(MouseEvent event) {
102    updateMouse(event);
103  }
104
105  /**
106   * You can ignore this method. This method gets called when the mouse
107   * exits the window, and is used internally.
108   */
109  public void mouseExited(MouseEvent event) {
110    Mouse.inWindow = false;
111  }
112  /**
113   * You can ignore this method. This method gets called when the mouse
114   * enters the window, and is used internally.
115   */
116  public void mouseEntered(MouseEvent event) {
117    Mouse.inWindow = true;
118  }
119  /**
120   * You can ignore this method. This method gets called when the mouse
121   * button is pressed, and is used internally.
122   */
123  public void mouseClicked(MouseEvent event) {}
124}
