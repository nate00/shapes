001package shapes;
002
003/**
004 * Represents a direction in the 2D plane. Directions can be used to control
005 * which way shapes face and which way they move.
006 * <p>
007 * Directions are represented in degrees (or radians) starting with 0 degrees
008 * pointing to the right, and increasing counterclockwise (like in a unit
009 * circle). For example, up is 90 degrees and left is 180 degrees.
010 */
011public class Direction {
012  private double degrees;
013
014  /**
015   * A direction pointing to the right (0 degrees).
016   */
017  public static Direction RIGHT = new Direction(0.0);
018  /**
019   * A direction pointing to the left (180 degrees).
020   */
021  public static Direction LEFT = new Direction(180.0);
022  /**
023   * A direction pointing downward (270 degrees).
024   */
025  public static Direction DOWN = new Direction(270.0);
026  /**
027   * A direction pointing upward (90 degrees).
028   */
029  public static Direction UP = new Direction(90.0);
030
031  /**
032   * Constructs a new direction pointing a given number of degrees
033   * counterclockwise from straight right. So, for example, 0 degrees points
034   * right, 90 degrees points up, and 180 degrees points left.
035   *
036   * @param degrees the number of degrees counterclockwise from straight
037   *                right this direction will point.
038   */
039  public Direction(double degrees) {
040    setDegrees(degrees);
041  }
042
043  /**
044   * Constructs a new direction pointing from one point to another. The
045   * constructed direction is the direction one would have to move to get from
046   * <code>start</code> to <code>finish</code>.
047   *
048   * @param start   where this direction starts.
049   * @param finish  where this direction ends.
050   */
051  public Direction(Point start, Point finish) {
052    if (start == null) {
053      throw new IllegalArgumentException("start must not be null");
054    }
055    if (finish == null) {
056      throw new IllegalArgumentException("finish must not be null");
057    }
058    double deltaX = finish.getX() - start.getX();
059    double deltaY = finish.getY() - start.getY();
060    double radians = Math.atan2(deltaY, deltaX);
061    setRadians(radians);
062  }
063
064  /**
065   * Returns a new direction pointing a given number of radians
066   * counterclockwise from straight right. So, for example, <code>0</code>
067   * radians points
068   * right, <code>Math.PI / 2.0</code> points up, and <code>Math.PI</code>
069   * points up.
070   *
071   * @param radians the number of radians counterclockwise from straight right
072   *                the returned direction will point.
073   * @return        a new direction pointing in the specified direction.
074   */
075  public static Direction inRadians(double radians) {
076    return new Direction(radians / Math.PI * 180.0);
077  }
078
079  /**
080   * Constructs a new direction pointing a given number of degrees
081   * counterclockwise from straight right. So, for example, 0 degrees points
082   * right, 90 degrees points up, and 180 degrees points left. (This method
083   * is basically identical to <code>new Direction(degrees)</code>.)
084   *
085   * @param degrees the number of degrees counterclockwise from straight
086   *                right the returned direction will point.
087   * @return        a new direction pointing in the specified direction.
088   */
089  public static Direction inDegrees(double degrees) {
090    return new Direction(degrees);
091  }
092
093  /**
094   * Return the number of radians this direction is from straight right when
095   * rotated counterclockwise. So, for example, if this direction points right,
096   * this method returns <code>0</code>, and if this direction points up it
097   * returns <code>Math.PI / 2.0</code>.
098   *
099   * @return  double  radians counterclockwise from straight right this
100   *                  direction points.
101   */
102  public double toRadians() {
103    return degrees / 180.0 * Math.PI;
104  }
105
106  /**
107   * Return the number of degrees this direction is from straight right when
108   * rotated counterclockwise. So, for example, if this direction points right,
109   * this method returns <code>0</code>, and if this direction points up it
110   * returns <code>90</code>.
111   *
112   *
113   * @return  double  degrees counterclockwise from straight right this
114   *                  direction points.
115   */
116  public double toDegrees() {
117    return degrees;
118  }
119
120  private void setDegrees(double degrees) {
121    degrees %= 360.0;
122    if (degrees < 0.0) {
123      degrees += 360.0;
124    }
125    this.degrees = degrees;
126  }
127
128  private void setRadians(double radians) {
129    double degrees = radians / Math.PI * 180.0;
130    setDegrees(degrees);
131  }
132
133  /**
134   * Returns the opposite direction. So, for example, if this direction
135   * points up, the returned direction will point down.
136   *
137   * @return  the opposite direction.
138   */
139  public Direction reverse() {
140    return new Direction(degrees + 180.0);
141  }
142
143  /**
144   * Returns a direction perpendicular to this one. Note that each direction
145   * in 2 dimensions has two perpendicular directions. This method returns
146   * the perpendicular direction that is a 90 degree counterclockwise rotation
147   * of this direction.
148   * <p>
149   * To get both perpendicular directions, use
150   * <code>direction.perpendicular()</code> and
151   * <code>direction.perpendicular().reverse()</code>.
152   *
153   * @return  a direction perpendicular to this one.
154   */
155  public Direction perpendicular() {
156    return new Direction(degrees + 90.0);
157  }
158
159  /**
160   * Returns a direction a given number of radians counterclockwise from this
161   * direction.
162   *
163   * @param radians the measure of the angle in radians between this direction
164   *                and the returned direction.
165   * @return        the direction <code>radians</code> radians counterclockwise
166   *                from this direction.
167   */
168  public Direction rotationByRadians(double radians) {
169    return Direction.inRadians(toRadians() + radians);
170  }
171
172  /**
173   * Returns a direction a given number of degrees counterclockwise from this
174   * direction. (This method is identical to {@link #rotation}.)
175   *
176   * @param degrees the measure of the angle in degrees between this direction
177   *                and the returned direction.
178   * @return        the direction <code>degrees</code> degrees counterclockwise
179   *                from this direction.
180   */
181  public Direction rotationByDegrees(double degrees) {
182    return new Direction(this.degrees + degrees);
183  }
184
185  /**
186   * Returns a direction a given number of degrees counterclockwise from this
187   * direction.
188   *
189   * @param degrees the measure of the angle in degrees between this direction
190   *                and the returned direction.
191   * @return        the direction <code>degrees</code> degrees counterclockwise
192   *                from this direction.
193   */
194  public Direction rotation(double degrees) {
195    return rotationByDegrees(degrees);
196  }
197
198  /**
199   * Returns a random direction. All directions are equally likely.
200   *
201   * @return  a random direction.
202   */
203  public static Direction random() {
204    return new Direction(Math.random() * 360.0);
205  }
206
207  /**
208   * Return whether this direction is more rightward than leftward.
209   *
210   * @return  <code>true</code> if this direction is within a quarter
211   *          revolution of straight right, <code>false</code> if not.
212   */
213  public boolean isRightward() {
214    return isWithinQuarterTurnOf(RIGHT);
215  }
216
217  /**
218   * Return whether this direction is more upward than downward.
219   *
220   * @return  <code>true</code> if this direction is within a quarter
221   *          revolution of straight up, <code>false</code> if not.
222   */
223  public boolean isUpward() {
224    return isWithinQuarterTurnOf(UP);
225  }
226
227  /**
228   * Return whether this direction is more leftward than rightward.
229   *
230   * @return  <code>true</code> if this direction is within a quarter
231   *          revolution of straight left, <code>false</code> if not.
232   */
233  public boolean isLeftward() {
234    return isWithinQuarterTurnOf(LEFT);
235  }
236
237  /**
238   * Return whether this direction is more downward than upward.
239   *
240   * @return  <code>true</code> if this direction is within a quarter
241   *          revolution of straight down, <code>false</code> if not.
242   */
243  public boolean isDownward() {
244    return isWithinQuarterTurnOf(DOWN);
245  }
246
247  boolean isWithinQuarterTurnOf(Direction dir) {
248    return
249      Math.abs(dir.toDegrees() - degrees) < 90.0 ||
250      Math.abs(dir.toDegrees() - degrees - 360.0) < 90.0 ||
251      Math.abs(dir.toDegrees() - degrees + 360.0) < 90.0;
252  }
253
254  /**
255   * Return whether this direction is equal to another object.
256   *
257   * @param obj the object with which to compare this direction.
258   * @return    <code>true</code> if <code>obj</code> is a
259   *            <code>Direction</code> object and it points in the same
260   *            direction as this one (within a very small tolerance).
261   */
262  @Override
263  public boolean equals(Object obj) {
264    if (obj instanceof Direction) {
265      return equals((Direction) obj);
266    }
267
268    return false;
269  }
270
271  private boolean equals(Direction that) {
272    return Geometry.equals(this.toDegrees(), that.toDegrees());
273  }
274
275  /**
276   * Returns a string representation of this direction.
277   *
278   * @return  a string representation of this direction.
279   */
280  @Override
281  public String toString() {
282    return degrees + " degrees";
283  }
284}
